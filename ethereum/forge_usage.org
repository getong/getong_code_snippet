* forge usage

** init

#+begin_src shell
forge init --offline hello_evm
#+end_src

** change the directory

#+begin_src shell
cd hello_evm
#+end_src

** remapping

#+begin_src shell
forge remappings
#+end_src

** build

#+begin_src shell
forge build
#+end_src

** create contract

#+begin_src shell
forge create \
  --broadcast \
  --rpc-url <YOUR_RPC_URL> \
  --private-key <YOUR_PRIVATE_KEY> \
  src/Counter.sol:Counter
#+end_src

** debug contract

#+begin_src shell
forge debug
#+end_src

** install library

#+begin_src shell
forge install username/contract-name
#+end_src

** forge inspect abi json file

#+begin_src shell
forge inspect --json your_solidity_filename.sol:ContractName abi
#+end_src

** forge script example

#+begin_src shell
forge script script/your_script.s.sol \
  --broadcast \
  --private-key your_private_key \
  --rpc-url endpoint_url
#+end_src

** cast call cli exmaple

#+begin_src shell
cast send 0xContractAddress "set(uint256)" 42 --private-key YOUR_PRIVAТE_KEY
cast call 0xContractAddress "get()"
cast balance <ACCOUNT ADDRESS>
#+end_src
