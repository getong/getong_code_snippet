** install foundry

#+begin_src shell
# enable proxy
$ curl -L https://foundry.paradigm.xyz | bash

# source env
$ source ~/.foundry/bin
# or
$ source ~/.zshenv

# install
$ foundryup

$ which anvil
~/.foundry/bin/anvil
#+end_src

** test with anvil

#+begin_src shell
